    Query Examples:

    • SELECT C.componentid,M.manufacturername, C.productnumber,C.category,PI.pricepaid AS [VALUE]
      FROM   manufacturer M
             INNER JOIN component C ON C.manufacturerid = M.manufacturerid
             INNER JOIN purchaseitem PI ON PI.componentid = C.componentid
             INNER JOIN purchaseorder PO ON PO.purchaseid = PI.purchaseid
      WHERE  Year(PO.orderdate) = '2003'
             AND PI.pricepaid = (SELECT Max(PI.pricepaid)
                                 FROM   purchaseitem PI
                                        INNER JOIN purchaseorder PO ON PI.purchaseid = PO.purchaseid
                                 WHERE  Year(PO.orderdate) = '2003') 

    • SELECT facilityid, version
      FROM   yssrepdb.dbo.yispollfile F
             INNER JOIN yssrepdb.dbo.yishosts H ON F.pollfileuid = H.pollfileuid
             INNER JOIN yssrepdb.dbo.yispackages P ON H.hostuid = P.hostuid
      WHERE  F.pollfileuid IN (SELECT pollfileuid
                               FROM   yssrepdb.dbo.yishosts
                               WHERE  hostuid IN (SELECT hostuid
                                                  FROM   yssrepdb.dbo.yispackages
                                                  WHERE  NAME = 'YUMPOS'))
             AND P.NAME = 'YUMPOS'
      GROUP  BY importdate,facilityid,version
      HAVING Max(importdate) = importdate
      ORDER  BY version DESC
